var $pop = Popcorn("#video-target"),
	image = document.createElement("image");

	image.id = "capture";
	image.style.marginLeft = "10px";

	$pop.listen("canplayall", function() {

		this.media.parentNode.appendChild( image );

		this.exec( 25, function() {

			this.pause().capture({

				target: "img#capture",
				media: true

			}).currentTime( 0 ).play();

		}).currentTime( 23 ).play();

	});
